DROP TABLE IF EXISTS `srampos_memberdiscountcards`;

CREATE TABLE `srampos_memberdiscountcards` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `card_details_id` int(11) NOT NULL,
